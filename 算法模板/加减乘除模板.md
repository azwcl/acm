## 1. 高精度除以低精度
```cpp
/**
 * a / b = c ... d
 */
vector<int> div(vector<int> &a, int b, int &c)
{
    vector <int> d;
    c = 0;// 余数是0
    for(int i = a.size() - 1; i >= 0; --i)
    {
        c = c*10 + a[i];
        d.push_back(c/b);
        c %= b;
    }
    reverse(d.begin(), d.end());

    while(d.size() > 1 && d.back() == 0) d.pop_back(); // 清除高位是0

    return d;
}
```
